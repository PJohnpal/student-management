import React, { useState, useEffect } from 'react';
import { ClipboardList, Plus, Search, User, BookOpen, Calendar } from 'lucide-react';
import LoadingSpinner from '../components/LoadingSpinner';
import Modal from '../components/Modal';
import { formatDate } from '../utils/helpers';

const Enrollments = () => {
  const [enrollments, setEnrollments] = useState([]);
  const [students, setStudents] = useState([]);
  const [courses, setCourses] = useState([]);
  const [loading, setLoading] = useState(true);
  const [searchTerm, setSearchTerm] = useState('');
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [formData, setFormData] = useState({
    student_id: '',
    course_id: '',
    status: 'active'
  });

  useEffect(() => {
    fetchData();
  }, []);

  const fetchData = async () => {
    try {
      setLoading(true);
      // Simulate API calls
      await new Promise(resolve => setTimeout(resolve, 1000));
      
      // Mock data
      setEnrollments([
        {
          id: 1,
          student_id: 1,
          course_id: 1,
          enrollment_date: '2024-01-15T00:00:00Z',
          status: 'active',
          student: {
            id: 1,
            student_id: 'S001',
            user: { full_name: 'John Doe' }
          },
          course: {
            id: 1,
            course_code: 'CS101',
            course_name: 'Introduction to Programming'
          }
        }
      ]);

      setStudents([
        { id: 1, student_id: 'S001', user: { full_name: 'John Doe' } },
        { id: 2, student_id: 'S002', user: { full_name: 'Jane Smith' } }
      ]);

      setCourses([
        { id: 1, course_code: 'CS101', course_name: 'Introduction to Programming' },
        { id: 2, course_code: 'MATH101', course_name: 'Calculus I' }
      ]);
    } catch (error) {
      console.error('Error fetching data:', error);
    } finally {
      setLoading(false);
    }
  };

  const filteredEnrollments = enrollments.filter(enrollment =>
    enrollment.student?.user?.full_name?.toLowerCase().includes(searchTerm.toLowerCase()) ||
    enrollment.course?.course_name?.toLowerCase().includes(searchTerm.toLowerCase()) ||
    enrollment.course?.course_code?.toLowerCase().includes(searchTerm.toLowerCase())
  );

  const handleInputChange = (e) => {
    setFormData({
      ...formData,
      [e.target.name]: e.target.value
    });
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    try {
      // Simulate API call
      await new Promise(resolve => setTimeout(resolve, 1000));
      console.log('Enrollment data:', formData);
      setIsModalOpen(false);
      setFormData({ student_id: '', course_id: '', status: 'active' });
      fetchData(); // Refresh data
    } catch (error) {
      console.error('Error creating enrollment:', error);
    }
  };

  if (loading) {
    return (
      <div className="container">
        <LoadingSpinner text="Loading enrollments..." />
      </div>
    );
  }

  return (
    <div className="container">
      <div style={{ padding: '2rem 0' }}>
        <div className="card-header">
          <h1>
            <ClipboardList size={24} style={{ marginRight: '0.5rem' }} />
            Enrollments Management
          </h1>
          <button 
            className="btn btn-success"
            onClick={() => setIsModalOpen(true)}
          >
            <Plus size={18} />
            New Enrollment
          </button>
        </div>

        {/* Search Bar */}
        <div className="card">
          <div style={{ position: 'relative' }}>
            <Search 
              size={20} 
              style={{
                position: 'absolute',
                left: '12px',
                top: '50%',
                transform: 'translateY(-50%)',
                color: '#64748b'
              }} 
            />
            <input
              type="text"
              placeholder="Search enrollments by student or course..."
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
              style={{
                width: '100%',
                padding: '0.75rem 1rem 0.75rem 2.5rem',
                border: '1px solid #d1d5db',
                borderRadius: '8px',
                fontSize: '1rem'
              }}
            />
          </div>
        </div>

        {/* Enrollments Table */}
        <div className="table-container">
          <table className="table">
            <thead>
              <tr>
                <th>Student</th>
                <th>Course</th>
                <th>Enrollment Date</th>
                <th>Status</th>
                <th>Actions</th>
              </tr>
            </thead>
            <tbody>
              {filteredEnrollments.map((enrollment) => (
                <tr key={enrollment.id}>
                  <td>
                    <div style={{ display: 'flex', alignItems: 'center', gap: '0.5rem' }}>
                      <User size={16} color="#64748b" />
                      <div>
                        <div style={{ fontWeight: '600' }}>
                          {enrollment.student?.user?.full_name}
                        </div>
                        <div style={{ fontSize: '0.875rem', color: '#64748b' }}>
                          ID: {enrollment.student?.student_id}
                        </div>
                      </div>
                    </div>
                  </td>
                  <td>
                    <div style={{ display: 'flex', alignItems: 'center', gap: '0.5rem' }}>
                      <BookOpen size={16} color="#64748b" />
                      <div>
                        <div style={{ fontWeight: '600' }}>
                          {enrollment.course?.course_code}
                        </div>
                        <div style={{ fontSize: '0.875rem', color: '#64748b' }}>
                          {enrollment.course?.course_name}
                        </div>
                      </div>
                    </div>
                  </td>
                  <td>
                    <div style={{ display: 'flex', alignItems: 'center', gap: '0.5rem' }}>
                      <Calendar size={16} color="#64748b" />
                      {formatDate(enrollment.enrollment_date)}
                    </div>
                  </td>
                  <td>
                    <span className={`badge badge-${enrollment.status === 'active' ? 'success' : 'warning'}`}>
                      {enrollment.status}
                    </span>
                  </td>
                  <td>
                    <div style={{ display: 'flex', gap: '0.5rem' }}>
                      <button className="btn btn-secondary" style={{ padding: '0.5rem' }}>
                        Edit
                      </button>
                      <button className="btn btn-danger" style={{ padding: '0.5rem' }}>
                        Drop
                      </button>
                    </div>
                  </td>
                </tr>
              ))}
              {filteredEnrollments.length === 0 && (
                <tr>
                  <td colSpan="5" style={{ textAlign: 'center', color: '#64748b', padding: '2rem' }}>
                    {searchTerm ? 'No enrollments found matching your search' : 'No enrollments found'}
                  </td>
                </tr>
              )}
            </tbody>
          </table>
        </div>

        {/* New Enrollment Modal */}
        <Modal
          isOpen={isModalOpen}
          onClose={() => {
            setIsModalOpen(false);
            setFormData({ student_id: '', course_id: '', status: 'active' });
          }}
          title="New Enrollment"
        >
          <form onSubmit={handleSubmit}>
            <div className="form-group">
              <label htmlFor="student_id">Student *</label>
              <select
                id="student_id"
                name="student_id"
                value={formData.student_id}
                onChange={handleInputChange}
                required
              >
                <option value="">Select a student</option>
                {students.map((student) => (
                  <option key={student.id} value={student.id}>
                    {student.student_id} - {student.user?.full_name}
                  </option>
                ))}
              </select>
            </div>

            <div className="form-group">
              <label htmlFor="course_id">Course *</label>
              <select
                id="course_id"
                name="course_id"
                value={formData.course_id}
                onChange={handleInputChange}
                required
              >
                <option value="">Select a course</option>
                {courses.map((course) => (
                  <option key={course.id} value={course.id}>
                    {course.course_code} - {course.course_name}
                  </option>
                ))}
              </select>
            </div>

            <div className="form-group">
              <label htmlFor="status">Status *</label>
              <select
                id="status"
                name="status"
                value={formData.status}
                onChange={handleInputChange}
                required
              >
                <option value="active">Active</option>
                <option value="completed">Completed</option>
                <option value="dropped">Dropped</option>
              </select>
            </div>

            <div style={{ display: 'flex', gap: '1rem', justifyContent: 'flex-end', marginTop: '1.5rem' }}>
              <button 
                type="button"
                className="btn btn-secondary"
                onClick={() => setIsModalOpen(false)}
              >
                Cancel
              </button>
              <button 
                type="submit" 
                className="btn btn-success"
              >
                Create Enrollment
              </button>
            </div>
          </form>
        </Modal>
      </div>
    </div>
  );
};

export default Enrollments;
